# Windows PowerShell
$env:DASHSCOPE_API_KEY="replace-with-your-key"; $env:HARNESS_HOST_PORT="14098"
docker run --rm -d `
  --name opencode-harness-threat-analysis `
  -e HARNESS_WEB=true `
  -e HARNESS_KEEP_WEB=true `
  -e "HARNESS_PUBLIC_URL=http://127.0.0.1:$($env:HARNESS_HOST_PORT)" `
  -e OPENCODE_MODEL=alibaba-cn/glm-5 `
  -e "OPENCODE_INITIAL_PROMPT=/threat-analysis 请扫描以下项目进行威胁分析，并严格使用给定的容器内绝对路径，输出格式和内容请严格遵照本skill;`n- 代码路径: /scan/project`n- 输出路径: /scan/output" `
  -e HARNESS_AUTH_PROVIDER=alibaba-cn `
  -e "HARNESS_AUTH_KEY=$env:DASHSCOPE_API_KEY" `
  -p "127.0.0.1:$($env:HARNESS_HOST_PORT):4096" `
  -v "${PWD}\workdirs\threat_analysis\entrypoint.sh:/entrypoint.sh:ro" `
  -v "${PWD}\workdirs\threat_analysis:/scan/opencode:ro" `
  -v "C:\absolute\path\to\project:/scan/project:ro" `
  -v "${PWD}\output\threat_analysis:/scan/output" `
  opencode-harness-image:base

# Linux Bash
export DASHSCOPE_API_KEY="replace-with-your-key" HARNESS_HOST_PORT="4098"
docker run --rm -d \
  --name opencode-harness-threat-analysis \
  -e HARNESS_WEB=true \
  -e HARNESS_KEEP_WEB=true \
  -e HARNESS_PUBLIC_URL="http://127.0.0.1:${HARNESS_HOST_PORT}" \

  -e HARNESS_AUTH_PROVIDER=alibaba-cn \
  -e HARNESS_AUTH_KEY="$DASHSCOPE_API_KEY" \
  -p "127.0.0.1:${HARNESS_HOST_PORT}:4096" \
  -v "$PWD/workdirs/threat_analysis/entrypoint.sh:/entrypoint.sh:ro" \
  -v "$PWD/workdirs/threat_analysis:/scan/opencode:ro" \
  -v "/absolute/path/to/project:/scan/project:ro" \
  -v "$PWD/output/threat_analysis:/scan/output" \
  opencode-harness-image:base
